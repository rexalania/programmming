#include <iostream>
using namespace std;

int main(){
	int num1;
	cout<<"enter a number: ";
	cin>>num1;
	cout<<"\nValue before being incremented: "<<num1<<"\n";
	num1++;
	cout<<"Value after being incremented: "<<num1<<"\n";
	num1--;
	cout<<"Value before being decremented: "<<num1<<"\n";
	num1--;
	cout<<"Value after being decrementes: "<<num1;
}
